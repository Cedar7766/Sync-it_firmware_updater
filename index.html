  <input type="file" id="hexFile" accept=".hex"><br>

  <button id="connect">Connect & Upload</button><br>

  <progress id="progress" value="0" max="100"></progress>


  <script src="avrbro.browser.js"></script>

  <script>

    const connBtn = document.getElementById('connect');

    const fileInput = document.getElementById('hexFile');

    const progressBar = document.getElementById('progress');


    connBtn.onclick = async () => {

      if (!fileInput.files.length) {

        alert("Please select a .hex file first.");

        return;

      }


      // 1) Open WebSerial, do 1200‑baud reset trick

      const port = await navigator.serial.requestPort();

      await port.open({ baudRate: 1200 });

      await new Promise(r => setTimeout(r, 250));

      await port.close();


      // 2) Re-open at 57600 to talk to bootloader

      await port.open({ baudRate: 57600 });

      const serial = new AvrSerial(port);

      await serial.open();


      const uploader = new STK500v1(serial);


      // 3) Read selected hex file

      const hexText = await fileInput.files[0].text();


      // 4) Flash

      try {

        await uploader.flashHex(hexText, pct => {

          progressBar.value = pct;

        });

        alert('Upload complete!');

      } catch (e) {

        alert('Upload failed: ' + e.message);

      } finally {

        await serial.close();

      }

    };

  </script>
